n = int(input("enter the number"))
q = n
rs = 0  
while q != 0:     
    r=q % 10       
    rs=(rs * 10) + r 
    q=q // 10
    
s = (n**n)
res = 0
c = 0
while s != 0:              #palindrome logic to check whether the last digits is same as the given number.
    rem = s % 10
    res = (res * 10) + rem
    if res == rs:
        c += 1
    s = s // 10
    
if c == 1:
    print("Automorphic number")
else:
    print("Not an Automorphic number")
